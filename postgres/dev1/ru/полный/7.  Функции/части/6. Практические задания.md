```sql
RETURNS TABLE(root1 double precision, root2 double precision) 
AS $$
  DECLARE
    d double precision := b * b - 4 * a * c;
  BEGIN
    IF d < 0 THEN
      RETURN;
    ELSIF d = 0 THEN
      RETURN QUERY SELECT -b / (2 * a), NULL;
    ELSE
      RETURN QUERY 
        SELECT (-b + sqrt(d)) / (2 * a), (-b - sqrt(d)) / (2 * a);
    END IF;
  END;
$$ LANGUAGE plpgsql;
```
